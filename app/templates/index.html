# 🐍 Ejemplo: Clase Estudiante con POO

class Estudiante:
    """Representa un estudiante con nombre y notas."""

    def __init__(self, nombre: str, carrera: str):
        self.nombre = nombre
        self.carrera = carrera
        self.notas: list[float] = []

    def agregar_nota(self, nota: float) -> None:
        """Agrega una nota a la lista del estudiante."""
        if 0 <= nota <= 10:
            self.notas.append(nota)

    def promedio(self) -> float:
        """Calcula el promedio de notas."""
        if not self.notas:
            return 0.0
        return round(sum(self.notas) / len(self.notas), 2)

    def __repr__(self) -> str:
        return f"Estudiante({self.nombre!r}, promedio={self.promedio()})"


# Uso de la clase
denis = Estudiante("Denis", "Ing. Software")
denis.agregar_nota(9.5)
denis.agregar_nota(8.8)
denis.agregar_nota(10.0)

print(denis.promedio())   # ✅ 9.43
print(repr(denis))        # ✅ Estudiante('Denis', promedio=9.43)
