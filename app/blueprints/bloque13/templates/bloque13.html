Bloque 13: Decoradores en Python

¿Qué es un decorador?

Un decorador es una función que recibe otra función, modifica su comportamiento y retorna una nueva función. Se aplica con @nombre_decorador antes de la función que se quiere decorar.

Estructura básica


def mi_decorador(funcion_original):
    def wrapper(*args, **kwargs):
        print("Antes de ejecutar")
        resultado = funcion_original(*args, **kwargs)
        print("Después de ejecutar")
        return resultado
    return wrapper

@mi_decorador
def saludar():
    print("Hola mundo")

saludar()
# Salida:
# Antes de ejecutar
# Hola mundo
# Después de ejecutar

Ejemplo práctico: validación


def validar_division(funcion):
    def wrapper(a, b):
        if b == 0:
            print("Error: división por cero")
            return None
        return funcion(a, b)
    return wrapper

@validar_division
def dividir(a, b):
    return a / b

print(dividir(10, 2))  # 5.0
print(dividir(10, 0))  # Error: división por cero

Ejercicios resueltos

Ejercicio 1: Decorador simple


def iniciar(funcion):
    def wrapper(*args, **kwargs):
        print("Iniciando ...")
        return funcion(*args, **kwargs)
    return wrapper

@iniciar
def hola():
    print("Hola Denis")

hola()
# Salida:
# Iniciando ...
# Hola Denis

Ejercicio 2: Decorador con validación


def positivo(funcion):
    def wrapper(x):
        if x <= 0:
            print("Error: debe ser positivo")
            return None
        return funcion(x)
    return wrapper

@positivo
def cuadrado(x):
    return x ** 2

print(cuadrado(4))   # 16
print(cuadrado(-3))  # Error: debe ser positivo

Ejercicio 3: Decorador de log


def log(funcion):
    def wrapper(*args, **kwargs):
        print("Llamando función ...")
        return funcion(*args, **kwargs)
    return wrapper

@log
def suma(a, b):
    return a + b

print(suma(2, 3))
# Salida:
# Llamando función ...
# 5
