Ejercicio 1: Relaciones UML Integradas

Identifica herencia, interfaz, agregación y dependencia dentro de un sistema de reservas.

hotel_reserva.py 
from abc import ABC, abstractmethod

class IGestion(ABC):

    @abstractmethod
    def registrar(self):
        pass

class Hotel:

    def __init__(self, nombre):
        self.nombre = nombre

class Persona:

    def __init__(self, cedula, nombre):
        self.cedula = cedula
        self.nombre = nombre

class Huesped(Persona):   # HERENCIA
    def __init__(self, cedula, nombre, telefono):
        super().__init__(cedula, nombre)
        self.telefono = telefono

class Habitacion:

    def __init__(self, numero, precio):
        self.numero = numero
        self.precio = precio


class DetalleReserva:

    def __init__(self, habitacion, dias):
        self.habitacion = habitacion
        self.dias = dias

    def subtotal(self):
        return self.habitacion.precio * self.dias


class Reserva(IGestion):   # INTERFAZ

    def __init__(self, huesped):   # AGREGACIÓN
        self.huesped = huesped
        self.detalles = []

    def agregar_habitacion(self, habitacion, dias):

        self.detalles.append(DetalleReserva(habitacion, dias))

    def registrar(self, hotel):

        print("Hotel:", hotel.nombre)
        print("Reserva registrada")

    def total(self):
        return sum(d.subtotal() for d in self.detalles)


h1 = Habitacion(101, 50)
h2 = Habitacion(202, 80)

cliente = Huesped("0912345678", "Carlos Pérez", "0999999999")

reserva = Reserva(cliente)

reserva.agregar_habitacion(h1, 2)
reserva.agregar_habitacion(h2, 1)

hotel = Hotel("Hotel Paraíso")

reserva.registrar(hotel)

print("Huésped:", reserva.huesped.nombre)
print("Total:", reserva.total())

class Persona:
    def __init__(self, cargo):
        self.cargo = cargo


class Administrador(Persona):
    def __init__(self, cargo):
        super().__init__(cargo)
        self.cargo = cargo

print("Es una relacion herencia porque administrador hereda de persona")
