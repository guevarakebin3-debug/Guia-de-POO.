Bloque 17: Mixins en Python

Concepto

Un Mixin es una clase auxiliar que no se usa sola. Se diseña para agregar funcionalidad a otras clases mediante herencia, evitando duplicar código. Por convención, su nombre suele terminar en Mixin.

Ejemplo básico


class ValidacionMixin:
    """Clase auxiliar - no se instancia sola."""

    def validar_nombre(self, nombre):
        if not nombre:
            raise ValueError("Nombre vacío")

    def validar_edad(self, edad):
        if edad < 0:
            raise ValueError("Edad inválida")

class SistemaEstudiantes(ValidacionMixin):
    def __init__(self):
        self.estudiantes = []

    def registrar(self, nombre, edad):
        self.validar_nombre(nombre)  # método del mixin
        self.validar_edad(edad)
        self.estudiantes.append({"nombre": nombre, "edad": edad})

sistema = SistemaEstudiantes()
sistema.registrar("Daniel", 20)
print(sistema.estudiantes)
# [{'nombre': 'Daniel', 'edad': 20}]

MRO - Orden de Resolución de Métodos

Python busca métodos en el orden de herencia (de izquierda a derecha). 


class A:
    def metodo(self): print("A")

class B:
    def metodo(self): print("B")

class C(A, B):
    pass

C().metodo()  # Imprime: A

Ejercicios resueltos

Ejercicio 1: Validar cédula


class ValidacionMixin:
    def validar_cedula(self, cedula):
        if len(cedula) != 10 or not cedula.isdigit():
            raise ValueError("Cédula inválida")

class SistemaEstudiantes(ValidacionMixin):
    def registrar(self, nombre, edad, cedula):
        self.validar_cedula(cedula)
        print(f"Estudiante {nombre} registrado con cédula {cedula}")

sistema = SistemaEstudiantes()
sistema.registrar("Ana", 22, "1234567890")

Ejercicio 2: Promedio con Mixin


class PromedioMixin:
    def calcular_promedio(self, notas):
        return sum(notas) / len(notas)

class Estudiante(PromedioMixin):
    def __init__(self, nombre, notas):
        self.nombre = nombre
        self.notas = notas

e = Estudiante("Luis", [10, 8, 9])
print(e.calcular_promedio(e.notas))  # 9.0
